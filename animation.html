//this is javascript folder const animatedDiv = document.getElementById('animated-div'); function moveElement() { let position = 0; const endPosition = window.innerWidth - animatedDiv.clientWidth; const animationSpeed = 5; // You can adjust the animation speed here (smaller value means faster animation). function animate() { if (position <= endPosition) { position += animationSpeed; animatedDiv.style.left = position + 'px'; requestAnimationFrame(animate); } else { position = 0; // Reset position after reaching the end of the screen animatedDiv.style.left = position + 'px'; requestAnimationFrame(animate); } } animate(); } moveElement();
